Public void insertionSort(int[] array ){
	for( var i = 1 ; i<array.length; i++){		// за всеки елемент се обхожда
		var current = array[i];         		// слага се текущ елемент за ключ
		j = i-1;                			// j е предходния (в ляво) елемент
while(j>=0 && array[j] > current){	// разглежда елементите един до друг
	array[j+1] = array[j]; 	 	//слага левия на мястото на десния
	j--;           	    //разглежда следвящия елемент(на ляво)
}
array[j+1] = current;          // слага разглеждания елемент(ключ) на позицията       (ако е последен/най-малък съответно най в ляво)
	}
}
